// Using While Loops

// declare/assign variables
var i = 1; // set counter to 1
var msg = ''; // message

// store times table in a variable
while (i < 10) {
    msg = i + ' x 5 = ' + (i * 5) + '<br />';
    i++;
}

// write the message into the page
document.getElementById("answer").innerHTML = msg;
