// Using Parameters With Event Handlers

var elUsername = document.getElementById('username'); // get username input
var elMsg = document.getElementById('feedback'); // get feedback element

function checkUsername() { // declare function
    if (this.value.length < 5) { // validate length of username
        elMsg.textContent = 'Username must be 5 characters or more'; // set msg
    } else { // otherwise
        elMsg.textContent = ''; // clear message
    }
}

elUsername.addEventHandler('blur', function() {  // anonymous function wraps checkUsername function
    checkUsername(5);  // allows checkUsername() to pass an argument
}, false);  // when it loses focus, call anonymous function passing argument 5 to function checkUsername(minLength)
