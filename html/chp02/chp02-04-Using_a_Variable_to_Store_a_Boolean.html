//Using a Variable to Store a Boolean

//declare variable
var inStock;
var shipping;

//assign variable: booleean value
inStock = true;
shipping = false;

//call/assign id attribute to display result to page
var elStock = document.getElementById("stock");
elStock.className = inStock ? "true" : "false"; //ternary operator
//if inStock is true, set class name to true, otherwise set to false

var elShip = document.getElementById("shipping");
elShip.className = shipping ? "true" : "false"; //ternary operator
//if shipping is true, set class name to true, otherwise set to false
