//Creating Objects Using Literal Notation
// creates object using the previous activities literal notation, representing a different hotel object than the last, like a different page of the same travel website

// Create an object called hotel with key value pairs for hotel name, number of total rooms, and rooms booked, and also a method that returns available rooms 
var hotel = {
    name: 'Quay',
    rooms: 40,
    booked: 25,
    checkAvailability: function() {
        return this.rooms - this.booked;
    }
};

//call/assign id "hotelName" to display hotel object's name result to page
var elName = document.getElementById("hotelName");
elName.textContent = hotel.name;

//call/assign id "rooms" to display hotel object's checkAvailability method result to page
var elRooms = document.getElementById("rooms");
elRooms.textContent = hotel.checkAvailability();
