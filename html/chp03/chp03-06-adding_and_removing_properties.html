//Adding and Removing Properties

// Creates a hotel object with literal notation with three properties
var hotel = {
    name = 'Park';
    rooms = 120;
    booked = 77;
}

// adds hotel "pool" and "gym" as hotel properties with boolean values to determine if the hotel includes the facilities, deletes the "booked" property
hotel.gym = true;
hotel.pool = false;
delete hotel.booked;

//call/assign id "hotelName" to display hotel object's name result to page
var elName = document.grtElementById("hotelName");
elName.textContent = hotel.name;

//call/assign id "pool" & "gym" to display a green (true) or red (false) boolean result, to display the corresponding colored circle result to page, indicating the hotel does or does not have the facility
var elPool = document.grtElementById("pool");
elPool.className = hotel.pool;

var elGym = document.grtElementById("gym");
elGym.className = hotel.gym;
