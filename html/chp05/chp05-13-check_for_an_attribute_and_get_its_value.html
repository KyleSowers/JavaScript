    <h1 id='header'>List King</h1>  //header image
    <h2>Buy groceries</h2>
    <ul>
        //list classes determine list item color, which are altered by JavaScript using the DOM
        <li id="one" class="hot">
            <em>fresh</em> figs
            </li>
        <li id="two" class="hot">pine nuts</li>
        <li id="three" class="hot">honey</li>
        <li id="four">balsamic vinegar</li>
    </ul>
    <div id="scriptResults"></div> //js: displays content
</div>
Show JavaScript 
// Check for an Attribute and Get Its Value

var firstItem = document.getElementById('one');// get the first list item

 // if it has a class attribute
if (firstItem.hasAttribute('class')) {
    var attr = firstItem.getAttribute('class'); // get the attribute

    // add the value of the attribute after the list
    var el = document.getElementById('scriptResults');
    el.innerHTML = '<p class="chp05-p">The first item has a class name: ' + attr + "</p>";
}
